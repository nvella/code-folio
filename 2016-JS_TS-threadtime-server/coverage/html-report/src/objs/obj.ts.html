// import { Promise } from 'es6-promise';
import * as mongodb from 'mongodb';
import * as shortid from 'shortid';
import * as _ from 'lodash';
import * as Joi from 'joi';
 
import { App, CollectionMap, ObjNotFoundError, ValidationError } from '../';
import { IObj, ObjType, ObjSchemaMap } from '@threadtime/threadtime-common';
 
// Generic object class
export class Obj implements IObj {
    private app: App;
 
    public _id: string;
    public owner_id: string;
    public parent_ids: string[];
    public time: number;
 
    private collection: mongodb.Collection;
 
    /**
     * Represents an Obj in Threadtime, server-side.
     * @param app Threadtime Server App instance
     * @param id Threadtime Obj ID
     */
    constructor(app: App, id?: string) {
        this.app = app;
 
        if(typeof id === 'string') {
            this._id = id;
        } else {
            this._id = `${ObjType.Generic}${shortid.generate()}`;
        }
 
        this.owner_id = null;
        this.parent_ids = [];
        this.time = Math.floor(Date.now() / 1000);
 
        this.collection = this.app.collectionFor(this.type);
    }
 
    get type(): ObjType {
        return parseInt(this._id[0]);
    }
 
    set type(objType: ObjType) {
        this._id = `${objType}${this._id.slice(1)}`;
    }
 
    get owner(): Obj {
        return new Obj(this.app, this.owner_id);
    }
 
    set owner(obj: Obj) {
        this.owner_id = obj._id;
    }
 
    get parents(): Obj[] {
        return this.parent_ids.map((id: string) => {
            return new Obj(this.app, id);
        });
    }
 
    /**
     * Add a parent given an Obj
     * @param parent Parent to add to the parents list
     */
    addParent(parent: Obj): void {
        this.parent_ids.push(parent._id);
    }
 
    /**
     * Remove a parent given an Obj
     * @param parent Parent to remove from the parents list
     */
    rmParent(parent: Obj): void {
        _.pull(this.parent_ids, parent._id);
    }
 
    /** Load the document from the relevant collection in the database
     *
     * @returns {Promise<void>|Promise} Promise resolved when document loaded, rejected on failure
     */
    load(): Promise<void> {
        return new Promise<void>((resolve, reject) => {
            this.collection.find({_id: this._id}).limit(1).toArray().then((docs: Object[]) => {
                if(_.isEmpty(docs)) return reject(new ObjNotFoundError());
                let doc = docs[0];
                _.forEach(doc, (val: any, key: string) => {
                    if(key.charAt(0) === '_') return; // Skip properties that begin with an underscore
                    (<any>this)[key] = val;
                });
                resolve();
            }).catch((err) => reject(err));
        });
    }
 
    validate(): IObjValidation {
        let schema = ObjSchemaMap[this.type];
        let result = Joi.validate(this, schema.joi, { allowUnknown: true });
 
        if(result.error === null) {
            return { success: true, error: null }
        } else {
            return { success: false, error: result.error }
        }
    }
 
    save(): Promise<void> {
        return new Promise<void>((resolve, reject) => {
            // Validate first
            let validation = this.validate();
            if(!validation.success) return reject(new ValidationError()); // TODO return with validation data
 
            let doc: {[x: string]: any} = {};
            for(let k of ObjSchemaMap[this.type].keys) doc[k] = (<any>this)[k];
 
            return this.collection.updateOne({_id: this._id}, doc, {upsert: true}).then(() => resolve());
        });
    }
}
 
export interface IObjValidation {
    success: boolean,
    error: Joi.ValidationError
}
