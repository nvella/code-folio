import * as sio from 'socket.io';
 
import { Message, HandshakeMessage, PermanentDisconnectMessage, DisconnectMachineReason, MessageType, NetHandler } from '@threadtime/threadtime-common';
import { App } from './app';
 
export class Client {
    private app: App; 
    private socket: SocketIO.Socket;
 
    netHandler: NetHandler;
    clientVersion: string | null;
    connected: boolean = false;
 
    constructor(app: App, socket: SocketIO.Socket) {
        this.app = app;
        this.socket = socket;
        this.netHandler = new NetHandler(socket);
    }
 
    start(): void {
        this.log('Starting connection...');
        // Emit 'client_disconnect' event on disconnect
        this.socket.on('disconnect', () => this.stop());
 
        // Wait for handshake message
        this.netHandler.onMessage(MessageType.Handshake, (msg: Message) => {
            let handshakeMsg = new HandshakeMessage().encapsulate(msg);
            this.log(`Received handshake v=${handshakeMsg.version}`);
 
            if(handshakeMsg.version !== App.VERSION) {
                this.log('Versions incompatible, kicking');
                let dcMsg = new PermanentDisconnectMessage();
                dcMsg.machineReason = DisconnectMachineReason.VersionMismatch;
                dcMsg.humanReason = `Client-Server version mismatch (server ${App.VERSION}`;
                this.netHandler.send(dcMsg);
                this.stop();
                return;
            }
 
            // Set client version in instance
            this.clientVersion = handshakeMsg.version;
 
            // Otherwise, reply back with handshake
            let resp = new HandshakeMessage();
            resp.version = App.VERSION;
            resp.replyTo = handshakeMsg.id;
            this.netHandler.send(resp);
 
            // TODO Emit 'handshake' event
        });
 
        this.connected = true;
    }
 
    stop(done: Function = () => {}): void {
        if(!this.connected) return done(false); // Don't 'disconnect' if already DC'd
 
        this.log('Stopping connection...');
        this.connected = false;
        if(this.socket.connected) this.socket.disconnect();
        this.app.emit('client_disconnect', this);
        return done();
    }
 
    log(str: string): void {
        this.app.log.info(`[${this.socket.client.conn.remoteAddress}] ${str}`);
    }
}
